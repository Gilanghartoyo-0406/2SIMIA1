                {
                if (document.f.inama.value=="")

                }

                {
                if (document.f.ialamat.value=="")

                {
                if ((document.f.inim.value!="")&&(document.f.inama.value!="")&& (document.f.ialamat.value!=""))

                }

                function timer()
                {
                var d = new Date();
                var jam = d.getHours();
                var menit = d.getMinutes(); var detik = d.getSeconds();
                var strwaktu = (jam<10)?"0"+jam:jam; strwaktu +=(menit<10)?"0"+menit:" : "+menit; strwaktu +=(detik<10)?"0"+detik:" : "+detik;
                document.f.txtwaktu.value=" "+strwaktu; setTimeout("timer()",200);
