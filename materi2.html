VARIABEL DALAM JAVA SCRIPT

1. penggunaan var, let, dan const 

          
              <!DOCTYPE html> 
              <html lang="en"> 
              <head>
                <meta charset="UTF-8">
                <title>Contoh Variabel JavaScript</title>
              </head>
              <body>
                <h2>Contoh Variabel: var, let, dan const</h2>
                <button onclick="contohVar()">Coba var</button>
                <button onclick="contohLet()">Coba let</button>
                <button onclick="contohConst()">Coba const</button>
                <p id="output"></p>

                <script>
                  function contohVar() {
                    var pesan = "Halo dari var!";
                    var pesan = "Var bisa dideklarasi ulang!";
                    document.getElementById("output").innerText = pesan;
                  }

                  function contohLet() {
                    let umur = 20;
                    umur = 21; // Bisa diubah
                    // let umur = 22; // ❌ Error kalau dibuka
                    document.getElementById("output").innerText = "Umur sekarang (let): " + umur;
                  }

                  function contohConst() {
                    const kota = "Jakarta";
                    // kota = "Bandung"; // ❌ Error kalau dibuka
                    document.getElementById("output").innerText = "Kota asal (const): " + kota;
                  }
                </script>
              </body>
              </html>

2. Menghitung Luas Persegi 

          
            <!DOCTYPE html>
            <html lang="en">
            <head>
              <meta charset="UTF-8">
              <title>Menghitung Luas Persegi Panjang</title>
            </head>
              <body>
                <h1>Hitung Luas Persegi Panjang</h1>

              <label for="panjang">Panjang (cm): </label>
              <input type="number" id="panjang"><br><br>

              <label for="lebar">Lebar (cm): </label>
              <input type="number" id="lebar"><br><br>

              <button onclick="hitungLuas()">Hitung Luas</button>

              <h2>Hasil:</h2>
              <p id="hasilLuas"></p>

              <script>
                function hitungLuas() {
                  // Ambil nilai dari input
                  let panjang = document.getElementById("panjang").value;
                  let lebar = document.getElementById("lebar").value;

                  // Hitung luas
                  const luas = panjang * lebar;

                  // Tampilkan hasil
                  document.getElementById("hasilLuas").innerText = 
                    `Luas persegi panjang adalah ${luas} cm²`;
                }
                </script>
              </body>
              </html>
