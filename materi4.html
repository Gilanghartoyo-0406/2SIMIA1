PROPERTI DAN METHOD DALAM CLASS

1. Deklarasi di dalam constructor() 

          
            class Mobil { 
                constructor(merk, warna) { 
                    this.merk = merk;   // Mendeklarasikan dan menginisialisasi properti 'merk' 
                    this.warna = warna; // Mendeklarasikan dan menginisialisasi properti 'warna' 
                    this.kecepatan = 0; // Mendeklarasikan dan menginisialisasi properti 'kecepatan' 
                dengan nilai default 
                } 
            } 
                
                let mobilSaya = new Mobil("Toyota", "Merah"); 
                console.log(mobilSaya.merk);     // Output: Toyota 
                console.log(mobilSaya.warna);    // Output: Merah 
                console.log(mobilSaya.kecepatan); // Output: 0

2. Deklarasi di luar constructor() (Field Declaration) 

          
                class Sepeda { 
                    merk = "Polygon"; // Deklarasi properti 'merk' dengan nilai default 
                    warna;           // Deklarasi properti 'warna' tanpa nilai default 
                    
                    constructor(warna) { 
                        this.warna = warna; // Menginisialisasi properti 'warna' di konstruktor 
                    } 
                } 
                    
                let sepedaSaya = new Sepeda("Biru"); 
                console.log(sepedaSaya.merk);  // Output: Polygon 
                console.log(sepedaSaya.warna); // Output: Biru 
                    
                let sepedaLain = new Sepeda(); 
                console.log(sepedaLain.merk);  // Output: Polygon 
                console.log(sepedaLain.warna); // Output: undefined (karena tidak diinisialisasi saat pembuatan objek)

3. Properti Instance Dinamis 

          
            let mobilLama = new Mobil("Daihatsu", "Hitam"); 
            mobilLama.tahunProduksi = 2005; // Menambahkan properti 'tahunProduksi' secara dinamis 
            ke objek mobilLama 
            console.log(mobilLama.tahunProduksi); // Output: 2005

4. Menambahkan Method (Fungsi dalam Class) 

          
            class Persegi { 
                constructor(sisi) { 
                this.sisi = sisi; 
            } 
            
            // Method untuk menghitung luas persegi 
            hitungLuas() { 
                return this.sisi * this.sisi; 
            } 
            
            // Method untuk menghitung keliling persegi 
            hitungKeliling() { 
                return 4 * this.sisi; 
            } 
            
            // Method yang mengakses properti lain dan memanggil method lain 
            deskripsi() { 
                console.log(`Persegi dengan sisi ${this.sisi}, luas ${this.hitungLuas()},  
                dan keliling ${this.hitungKeliling()}.`); 
            } 
            } 
            
            let persegiA = new Persegi(5); 
            console.log(persegiA.hitungLuas());    // Output: 25 
            console.log(persegiA.hitungKeliling()); // Output: 20 
            persegiA.deskripsi();                  // Output: Persegi dengan sisi 5, luas 25, dan keliling 20.

5. Mengakses Properti dari Dalam Class 

          
            class Orang { 
                constructor(nama, usia) { 
                    this.nama = nama; 
                    this.usia = usia; 
                } 
                
                ucapkanSalam() { 
                    console.log(`Halo, nama saya ${this.nama}.`); // Mengakses properti 'nama' 
                menggunakan 'this' 
                } 
                
                tambahUsia(tahun) { 
                    this.usia += tahun; // Mengubah nilai properti 'usia' menggunakan 'this' 
                } 
                
                lihatUsia() { 
                    console.log(`Usia saya sekarang ${this.usia} tahun.`); // Mengakses properti 
                'usia' menggunakan 'this' 
                } 
            } 
                
            let orang1 = new Orang("Alice", 25); 
            orang1.ucapkanSalam(); // Output: Halo, nama saya Alice. 
            orang1.tambahUsia(2); 
            orang1.lihatUsia();  // Output: Usia saya sekarang 27 tahun.

6. Mengakses Method Lain dari Dalam Class 

          
            class Kalkulator { 
                constructor(angka1, angka2) { 
                    this.angka1 = angka1; 
                    this.angka2 = angka2; 
                } 
                
                tambah() { 
                    return this.angka1 + this.angka2; 
                } 
                
                kali() { 
                    return this.angka1 * this.angka2; 
                } 
                
                hasilOperasi() { 
                    // Memanggil method 'tambah' menggunakan 'this' 
                    console.log(`Hasil penjumlahan: ${this.tambah()}`);  
                
                    // Memanggil method 'kali' menggunakan 'this' 
                    console.log(`Hasil perkalian: ${this.kali()}`);    
                } 
            } 
                
            let calc = new Kalkulator(10, 5); 
            calc.hasilOperasi(); 
            // Output: Hasil penjumlahan: 15 
            // Output: Hasil perkalian: 50
