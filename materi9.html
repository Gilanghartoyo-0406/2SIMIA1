2. Perulangan While

    
<html>
<head>
  <title>Latihan Perulangan II</title>
</head>
<script language="JavaScript">
  var deret = prompt('Masukkan Jumlah Deret :','');
  document.writeln("<h2>Perulangan Dengan While</h2>");
  document.writeln(" ");
  document.writeln("<br>");
  var jml = 0.0;
  var angka = 1;
  while (angka <= deret) {
    jml = jml + angka;
    angka++;
  }
  document.writeln("<br>");
  document.writeln("Jumlah Deret dari 1 sampai " + deret + " adalah = " + jml);
  document.writeln("<br>");
  document.writeln(" ");
</script>
<body>
</body>
</html>

3. Perulangan Do While

    
<!DOCTYPE html>
<html>
<head>
  <title>Simulasi Do While dengan Form</title>
</head>
<body>
  <h2>Simulasi Perulangan Do...While</h2>
  <form onsubmit="tampilkanPerulangan(); return false;">
    <label>Masukkan angka maksimal: </label>
    <input type="number" id="angka" min="1" required>
    <input type="submit" value="Mulai Perulangan">
  </form>

  <hr>
  <div id="hasil"></div>

  <script>
    function tampilkanPerulangan() {
      let batas = parseInt(document.getElementById("angka").value);
      let i = 1;
      let output = "";
      do {
        output += "Perulangan ke-" + i + "<br>";
        i++;
      } while (i <= batas);
      document.getElementById("hasil").innerHTML = output;
    }
  </script>
</body>
</html>
