Class dan object

1. Contoh penggunaan class dan object 

          
            class Mahasiswa { 
                constructor(nama, nim) { 
                    this.nama = nama; // menyimpan nilai ke properti 'nama' 
                    this.nim = nim;   // menyimpan nilai ke properti 'nim'  
                } 
                perkenalan() { 
                    console.log(`Halo, saya ${this.nama}, NIM ${this.nim}`); 
                } 
            }
            
            // Membuat objek mahasiswa1 dengan nama "Budi" dan NIM "12345" 
            let mahasiswa1 = new Mahasiswa("Budi", "12345"); 

            // Membuat objek mahasiswa2 dengan nama "Siti" dan NIM "67890" 
            let mahasiswa2 = new Mahasiswa("Siti", "67890"); 

            // Mengakses properti objek 
            console.log(mahasiswa1.nama); // Output: Budi 
            console.log(mahasiswa2.nim);  // Output: 67890 

            // Memanggil metode perkenalan pada objek 
            mahasiswa1.perkenalan(); // Output: Halo, saya Budi, NIM 12345 
            mahasiswa2.perkenalan(); // Output: Halo, saya Siti, NIM 67890

2. Implementasi dasar class dan object 

          
            // Definisi Class (Cetak Biru) 
                class Mobil { 
                constructor(merk, warna) { 
                    this.merk = merk; 
                    this.warna = warna; 
                    this.kecepatan = 0; 
                } 
                
                maju() { 
                    this.kecepatan += 10; 
                    console.log(`${this.merk} berwarna ${this.warna} melaju dengan kecepatan ${this.kecepatan} km/jam.`); 
                } 
                
                berhenti() { 
                    this.kecepatan = 0; 
                    console.log(`${this.merk} berwarna ${this.warna} berhenti.`); 
                } 
                } 
                
                // Membuat Objek (Membangun Mobil dari Cetak Biru) 
                let mobilSaya = new Mobil("Honda", "Merah"); // Memanggil konstruktor dengan merk 
                "Honda" dan warna "Merah" 

                let mobilTeman = new Mobil("Toyota", "Hitam"); // Membuat objek lain dengan merk 
                "Toyota" dan warna "Hitam" 
                
                // Mengakses Properti Objek 
                console.log(mobilSaya.merk);   // Output: Honda 
                console.log(mobilTeman.warna);  // Output: Hitam 
                
                // Memanggil Metode Objek 
                mobilSaya.maju();    // Output: Honda berwarna Merah melaju dengan kecepatan 10 km/jam. 
                mobilTeman.berhenti(); // Output: Toyota berwarna Hitam berhenti.
