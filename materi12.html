2. Program JavaScript (Validasi Form)

    
        <Html>
        <Body>
        <SCRIPT LANGUAGE="JavaScript">
        function isiform(form)
        {
        // Memanggil fungsi validasi dan memeriksa hasilnya
        var namaValid = isinama(form);
        var emailValid = isiemail(form);
        var komentarValid = isikomentar(form);

        if (namaValid && emailValid && komentarValid) {
            // Jika semua valid, submit form
            form.submit(); // Menggunakan form.submit() bukan form.submit;
        } else {
            // Jika ada yang tidak valid, tampilkan pesan kesalahan
            salahisi(form);
        }
        }

        // Fungsi kosongkan di sini tidak relevan lagi dengan cara isiform() di atas
        // function kosongkan(form) { ... }

        function salahisi(form)
        {
        var teks ="Ada Kesalahan Isian :";
        if (!isinama(form)) // Menggunakan ! untuk memeriksa jika false
        {teks +="\nNama Anda";}
        if (!isiemail(form))
        {teks +="\nEmail Anda";}
        if (!isikomentar(form))
        {teks +="\nKomentar Anda";}
        alert(teks);
        }

        function isinama(form)
        {
        if (form.nama.value === "") // Menggunakan === untuk perbandingan ketat
        {return false;}
        else
        {return true;}
        }

        function isiemail(form)
        {
        if((form.email.value === "" || form.email.value.indexOf('@',0) === -1) || form.email.value.indexOf('.') === -1)
        {return false;}
        else
        {return true;}
        }

        function isikomentar(form)
        {
        if(form.cs.value === "")
        {return false;}
        else
        {return true;}
        }
        </SCRIPT>
