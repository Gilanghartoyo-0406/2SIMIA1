2. (array2.html)

        
            <!DOCTYPE html>
            <html>
            <head>
            <title>Form Cari Nama dengan Sorot</title>
            <style>
            .highlight {
                color: red;
                font-weight: bold;
            }
            </style>
            </head>
            <body>
            <h2>Form Tambah Nama</h2>
            <label>Masukkan Nama :</label> <input type="text" id="namaBaru">
            <button type="button" onclick="tambahNama()">Tambah</button>
            <button type="button" onclick="tampilkanSemua()">Tampilkan Semua</button>

            <br><br>

            <label>Cari Nama :</label> <input type="text" id="namaCari">
            <button type="button" onclick="cariNama()">Cari Index</button>

            <hr>

            <div id="output" style="white-space: pre-line; font-family: monospace;"></div>

            <script>
            let daftarNama = [];
            let highlightIndex = -1; // untuk penanda index yang disorot

            function tambahNama() {
                const input = document.getElementById("namaBaru").value.trim();
                if (input === "") {
                    alert("Nama tidak boleh kosong.");
                    return;
                }
                daftarNama.push(input);
                document.getElementById("namaBaru").value = "";
                highlightIndex = -1; // reset sorotan
                alert("Nama berhasil ditambahkan!");
            }

            function tampilkanSemua() {
                const output = document.getElementById("output");
                if (daftarNama.length === 0) {
                    output.innerHTML = "Belum ada nama yang ditambahkan.";
                    return;
                }
                let teks = "";
                for (let i = 0; i < daftarNama.length; i++) {
                    if (i === highlightIndex) {
                        teks += `Index ke-${i}: <span class="highlight">${daftarNama[i]}</span>\n`;
                    } else {
                        teks += `Index ke-${i}: ${daftarNama[i]}\n`;
                    }
                }
                output.innerHTML = teks;
            }

            function cariNama() {
                const dicari = document.getElementById("namaCari").value.trim();
                if (dicari === "") {
                    alert("Silakan isi nama yang ingin dicari.");
                    return;
                }
                const index = daftarNama.indexOf(dicari);
                if (index !== -1) {
                    highlightIndex = index; // simpan index yang akan disorot
                    tampilkanSemua(); // refresh tampilan
                    alert(`Nama "${dicari}" ditemukan di index ke-${index}.`);
                } else {
                    highlightIndex = -1;
                    tampilkanSemua();
                    alert(`Nama "${dicari}" tidak ditemukan.`);
                }
            }
            </script>
            </body>
            </html>

3. Fungsi tentang Tanggal dan Waktu (tanggal.html)

    
        <!DOCTYPE html>
        <html>
        <head>
        <title>Fungsi Tanggal dengan Form</title>
        </head>
        <body>
        <h2>Form Pilih Tanggal</h2>
        <form name="formTanggal">
        <label for="tgl">Pilih Tanggal:</label>
        <input type="date" name="tanggal" id="tgl">
        <input type="button" value="Tampilkan Info Tanggal" onclick="tampilkanTanggal()">
        </form>

        <p id="hasil"></p>

        <script>
        function tampilkanTanggal() {
            var form = document.forms["formTanggal"];
            var tglInput = form["tanggal"].value;

            if (tglInput === "") {
                document.getElementById("hasil").innerHTML = "Silakan pilih tanggal terlebih dahulu.";
                return;
            }

            var tgl = new Date(tglInput);

            var namaHari = ["Minggu", "Senin", "Selasa", "Rabu", "Kamis", "Jumat", "Sabtu"];
            var namaBulan = [
                "Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus",
                "September", "Oktober", "November", "Desember"
            ];

            var hari = namaHari[tgl.getDay()];
            var tanggal = tgl.getDate();
            var bulan = namaBulan[tgl.getMonth()];
            var tahun = tgl.getFullYear();

            var hasil = "Tanggal dipilih: " + tanggal + " " + bulan + " " + tahun + "<br>";
            hasil += "Hari: " + hari;

            document.getElementById("hasil").innerHTML = hasil;
        }
        </script>
        </body>
        </html>
