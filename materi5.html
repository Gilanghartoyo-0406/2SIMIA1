2. Inisialisasi Properti Dasar 

          
                class Rectangle { 
                    constructor(width, height) { 
                        this.width = width; 
                        this.height = height; 
                        this.area = this.width * this.height; // Properti dapat dihitung di konstruktor 
                    } 
                    
                    getArea() { 
                        return this.area; 
                    } 
                } 
                    
                let rect1 = new Rectangle(5, 10); 
                console.log(rect1.width);  // Output: 5 
                console.log(rect1.height); // Output: 10 
                console.log(rect1.area);   // Output: 50 
                console.log(rect1.getArea()); // Output: 50

3. Properti dengan Nilai Default 

          
           class Circle { 
                constructor(radius = 1) { // Jika radius tidak diberikan, defaultnya 1 
                    this.radius = radius; 
                    this.diameter = 2 * this.radius; 
                } 
                
                getDiameter() { 
                    return this.diameter; 
                } 
            } 
                
            let circle1 = new Circle(7); 
            console.log(circle1.radius);   // Output: 7 
            console.log(circle1.diameter); // Output: 14 
                
            let circle2 = new Circle(); // Tidak memberikan argumen, menggunakan nilai default 
            console.log(circle2.radius);   // Output: 1 
            console.log(circle2.diameter); // Output: 2

4. Validasi Input dalam Konstruktor 

          
            class Product { 
                constructor(name, price) { 
                    if (typeof name !== 'string' || name.trim() === '') { 
                        throw new Error("Nama produk harus berupa string yang tidak kosong."); 
                    } 
                    if (typeof price !== 'number' || price <= 0) { 
                        throw new Error("Harga produk harus berupa angka positif."); 
                    } 
                    this.name = name; 
                    this.price = price; 
                } 
                
                displayInfo() { 
                    console.log(`${this.name}: $${this.price.toFixed(2)}`); 
                } 
            } 
                
                try { 
                let product1 = new Product("Laptop", 1200); 
                product1.displayInfo(); // Output: Laptop: $1200.00 
                let product2 = new Product("Mouse", -5); // Akan melempar error 
                } catch (error) { 
                console.error("Terjadi kesalahan saat membuat produk:", error.message); 
                }

5. Menambahkan Method (Fungsi dalam Class) 

          
            class Persegi { 
                constructor(sisi) { 
                this.sisi = sisi; 
            } 
            
            // Method untuk menghitung luas persegi 
            hitungLuas() { 
                return this.sisi * this.sisi; 
            } 
            
            // Method untuk menghitung keliling persegi 
            hitungKeliling() { 
                return 4 * this.sisi; 
            } 
            
            // Method yang mengakses properti lain dan memanggil method lain 
            deskripsi() { 
                console.log(`Persegi dengan sisi ${this.sisi}, luas ${this.hitungLuas()},  
                dan keliling ${this.hitungKeliling()}.`); 
            } 
            } 
            
            let persegiA = new Persegi(5); 
            console.log(persegiA.hitungLuas());    // Output: 25 
            console.log(persegiA.hitungKeliling()); // Output: 20 
            persegiA.deskripsi();                  // Output: Persegi dengan sisi 5, luas 25, dan keliling 20.
